BEGIN;

-- ==========================================================
-- V121 안전한 물품 삭제
-- 구매신청 연결, 현재재고, 입출고 이력이 있으면 삭제하지 않음
-- ==========================================================

DROP FUNCTION IF EXISTS public.admin_delete_inventory_item_safe(text);

CREATE FUNCTION public.admin_delete_inventory_item_safe(
  p_item_id text
)
RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_item public.inventory_items%ROWTYPE;
  v_admin boolean := false;
  v_purchase_count integer := 0;
  v_history_count integer := 0;
BEGIN
  IF auth.uid() IS NULL THEN
    RAISE EXCEPTION '로그인이 필요합니다.';
  END IF;

  SELECT
    COALESCE(p.is_super_admin,false)
    OR TRIM(COALESCE(p.name,'')) = '손동오'
    OR TRIM(COALESCE(p.emp_no::text,'')) = '201911041'
  INTO v_admin
  FROM public.profiles p
  WHERE p.id::text = auth.uid()::text
  LIMIT 1;

  IF NOT COALESCE(v_admin,false) THEN
    RAISE EXCEPTION '관리자만 품목을 삭제할 수 있습니다.';
  END IF;

  SELECT *
  INTO v_item
  FROM public.inventory_items
  WHERE id::text = TRIM(p_item_id)
  FOR UPDATE;

  IF NOT FOUND THEN
    RAISE EXCEPTION '삭제할 품목을 찾을 수 없습니다.';
  END IF;

  SELECT COUNT(*)
  INTO v_purchase_count
  FROM public.purchase_requests
  WHERE inventory_item_id::text = v_item.id::text;

  SELECT COUNT(*)
  INTO v_history_count
  FROM public.inventory_transactions
  WHERE item_id::text = v_item.id::text;

  IF v_purchase_count > 0 THEN
    RAISE EXCEPTION '구매신청에 연결된 품목이라 삭제할 수 없습니다.';
  END IF;

  IF COALESCE(v_item.current_stock,0) <> 0 THEN
    RAISE EXCEPTION '현재 재고가 남아 있어 삭제할 수 없습니다. 재고를 0으로 조정하세요.';
  END IF;

  IF v_history_count > 0 THEN
    RAISE EXCEPTION '입출고 이력이 있는 품목이라 삭제할 수 없습니다.';
  END IF;

  DELETE FROM public.inventory_items
  WHERE id::text = v_item.id::text;

  RETURN jsonb_build_object(
    'success',true,
    'item_id',v_item.id,
    'message','사용 기록이 없는 품목을 삭제했습니다.'
  );
END;
$$;

REVOKE ALL
ON FUNCTION public.admin_delete_inventory_item_safe(text)
FROM public;

GRANT EXECUTE
ON FUNCTION public.admin_delete_inventory_item_safe(text)
TO authenticated;

GRANT SELECT
ON TABLE public.purchase_requests,
         public.inventory_transactions,
         public.inventory_items
TO authenticated;

NOTIFY pgrst, 'reload schema';

COMMIT;
